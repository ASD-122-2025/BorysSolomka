def insertion_sort(nums):
    for j in range(1, len(nums)):
        key = nums[j]
        i = j - 1
        while i >= 0 and nums[i] > key:
            nums[i + 1] = nums[i]
            i -= 1
        nums[i + 1] = key
    return nums

def bubble_sort(nums):
    n = len(nums)
    for i in range(n):
        swapped = False
        for j in range(0, n-i-1):
            if nums[j] > nums[j+1]:
                nums[j], nums[j+1] = nums[j+1], nums[j]
                swapped = True
        if not swapped:
            break
    return nums

def measure_time(sort_function, array):
    arr_copy = array.copy()
    start_time = time.time()
    sort_function(arr_copy)
    end_time = time.time()
    return end_time - start_time

sizes = [5, 10, 50, 100, 500, 1000, 2000, 3000, 4000, 5000, 10000]

insertion_times = {}
bubble_times = {}

for size in large_sizes:
    random_array = [random.randint(-1000, 1000) for _ in range(size)]
    insertion_times[size] = measure_time(insertion_sort, random_array)
    
    if size <= 10000:
        bubble_times[size] = measure_time(bubble_sort, random_array)
    
    print(f" : {size}")
    print(f"   : {insertion_times[size]:.6f} .")
    if size <= 10000:
        print(f"   : {bubble_times[size]:.6f} .")
    print("---")
